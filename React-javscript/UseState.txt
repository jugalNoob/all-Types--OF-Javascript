import React, { useState } from 'react';

function App() {
  const [ch, setCh] = useState('jugal sharma');

  const handleClick = () => {
    setCh(prevCh => prevCh === 'jugal sharma' ? 'kanaika sharma' : 'jugal sharma');
  };

  return (
    <div>
      <h1>{ch}</h1>
      <button onClick={handleClick}>Click me!</button>
    </div>
  );
}

export default App;
